Quick Start (curl)

      
# Start a crawl
curl -X POST {{ request.url.scheme }}://{{ request.url.netloc }}/api/v1/crawl \
  -H "Content-Type: application/json" \
  -d '{"url":"https://example.com"}'

# Extract content from a single page
curl -X POST {{ request.url.scheme }}://{{ request.url.netloc }}/api/v1/extract \
  -H "Content-Type: application/json" \
  -d '{"url":"https://example.com"}'

# Check crawl status
curl {{ request.url.scheme }}://{{ request.url.netloc }}/api/v1/status/JOB_ID
